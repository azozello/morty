NumberParser

public class NumberParser{

StringBuilder stringBuilder = new StringBuilder();
int number;

public NumberParser(int tenNumber){
this.number = tenNumber;
}

int parseTenIn(int ten,int whatSystem){
rec(ten,whatSystem);
return Integer.parseInt(stringBuilder.reverse().toString());
}

private void rec(int n, int whatSystem){
if (n<whatSystem){
stringBuilder.append(n);
}
else{
stringBuilder.append(""+n%whatSystem);
rec(n/whatSystem,whatSystem);
}
}

@Override
public String toString() {
return String.valueOf(number);
}

}
